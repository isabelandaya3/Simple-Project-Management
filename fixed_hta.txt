        var ITEM_CONFIG = {
            item_id: 43,
            item_type: "Submittal",
            identifier: "Submittal #270",
            title: "LEBADMIN_061543_Calc_Area AB_Admin CLT Roof Structural Calculations & Gravity Beam Analysis",
            token: "78nfbdCM11ce2HebPLJxGtRak1ViUVbNIDlt5_1q_nY",
            reviewer_name: "Iyer, Padmanabhan",
            reviewer_email: "Padmanabhan.Iyer@hdrinc.com",
            folder_path: "\\\\sac-filsrv1\\Projects\\Structural-028\\Projects\\LEB\\9.0_Const_Svcs\\Mortenson\\Submittals\\Submittal - 270 - LEBADMIN_061543_Calc_Area AB_Admin CLT Roof Structural Calculations & Gravity Beam Analysis",
            is_rfi: false
        };
        
        var fso = new ActiveXObject("Scripting.FileSystemObject");
        var shell = new ActiveXObject("WScript.Shell");
        
        // Get folder path from the HTML element (avoids JS escaping issues)
        function getFolderPath() {
            return document.getElementById("folder-path").innerText;
        }

        // Initialize on load
        window.onload = function() {
            loadFolderFiles();
            // Show RFI question section if this is an RFI
            if (ITEM_CONFIG.is_rfi) {
                var rfiSection = document.getElementById("rfi-question-section");
                if (rfiSection) {
                    rfiSection.style.display = "block";
                }
            }
        };
        
        function openFolder() {
            var folderPath = getFolderPath();
            if (fso.FolderExists(folderPath)) {
                shell.Run('explorer.exe "' + folderPath + '"', 1, false);
            } else {
                alert("Folder not found: " + folderPath);
            }
        }
        
        function loadFolderFiles() {
            var folderPath = getFolderPath();
            var fileListEl = document.getElementById("file-list");
            
            if (!fso.FolderExists(folderPath)) {
                fileListEl.innerHTML = "<p style='color: #dc3545; padding: 8px;'>Folder not found</p>";
                return;
            }
            
            // Use DIR command to list PDFs - works better with special characters in paths
            var exec = shell.Exec('cmd /c dir /b "' + folderPath + '\\*.pdf"');
            var output = exec.StdOut.ReadAll();
            var html = "";
            var count = 0;
            
            if (output && output.length > 0) {
                var fileNames = output.split("\r\n");
                for (var i = 0; i < fileNames.length; i++) {
                    var fileName = fileNames[i].replace(/^\s+|\s+$/g, ''); // trim
                    if (fileName.length > 0) {
                        html += "<div class='file-item'>" +
                            "<input type='checkbox' name='selected_files' value='" + fileName.replace(/'/g, "&#39;") + "' id='file_" + count + "' checked>" +
                            "<label for='file_" + count + "'>" + fileName + "</label>" +
                            "</div>";
                        count++;
                    }
                }
            }
            
            if (count === 0) {
                html = "<p style='color: #888; padding: 8px; text-align: center;'>No PDF files in folder</p>";
            }
            
            fileListEl.innerHTML = html;
        }

        function submitResponse() {
            var category = document.getElementById('response_category').value;
            var notes = document.getElementById('notes').value;
            var internalNotes = document.getElementById('internal_notes').value;
            
            if (!category) {
                alert('Please select a response category.');
                return;
            }
            
            // Gather checked files
            var checkedFiles = [];
            var checkboxes = document.querySelectorAll('input[name="selected_files"]:checked');
            for (var i = 0; i < checkboxes.length; i++) {
                checkedFiles.push(checkboxes[i].value);
            }
            
            // Build response object
            var response = {
                _form_type: "reviewer_response",
                _version: "1.0",
                _submitted_at: new Date().toISOString(),
                item_id: ITEM_CONFIG.item_id,
                identifier: ITEM_CONFIG.identifier,
                token: ITEM_CONFIG.token,
                reviewer_name: ITEM_CONFIG.reviewer_name,
                reviewer_email: ITEM_CONFIG.reviewer_email,
                response_category: category,
                selected_files: checkedFiles,
                notes: notes,
                internal_notes: internalNotes
            };
            
            var jsonString = JSON.stringify(response, null, 2);
            
            // Save to Responses folder (same folder as this HTA file)
            // Get the folder from ITEM_CONFIG since document.location may not preserve UNC paths
            var responsesFolder = ITEM_CONFIG.folder_path + "\\Responses";
            var filePath = responsesFolder + "\\_reviewer_response.json";
            
            try {
                // Ensure Responses folder exists
                if (!fso.FolderExists(responsesFolder)) {
                    fso.CreateFolder(responsesFolder);
                }
                
                var file = fso.CreateTextFile(filePath, true, false);
                file.Write(jsonString);
                file.Close();
                
                // Success!
                document.getElementById('response-form').className = 'hidden';
                document.getElementById('success-message').innerHTML = 
                    '<h3>Response Submitted Successfully!</h3>' +
                    '<p style="margin-top: 10px;">Your response has been saved to the Responses folder.</p>' +
                    '<p style="margin-top: 5px; font-size: 13px;">The tracker will automatically import it when it runs.</p>' +
                    '<button class="open-folder-btn" onclick="openResponsesFolder()" style="margin-top: 15px;">View in Folder</button>';
                document.getElementById('success-message').className = 'success-box';
            } catch (e) {
                alert("Error saving response: " + e.message + "\\nPath: " + filePath);
            }
        }
        
        function openResponsesFolder() {
            var responsesFolder = ITEM_CONFIG.folder_path + "\\Responses";
            shell.Run('explorer.exe "' + responsesFolder + '"', 1, false);
        }
